Future<Quest> getdailyQuest(City city) async {
  AppLogger().log('QuestService: getdailyQuest() f端r Stadt ${city.name} (ID: ${city.id})');
  final response = await api.get('dailyquest/${city.id}');
  if (response.statusCode == 200){
    final List<dynamic> jsonList = jsonDecode(response.body);
    if (jsonList.isEmpty) {
      AppLogger().log('QuestService: Keine Daily Quest f端r Stadt ${city.name} gefunden');
      throw Exception('Keine Daily Quest f端r Stadt ${city.id} gefunden.');
    }
    final Map<String, dynamic> json = jsonList[0];
    final quest = Quest.fromJson(json);
    AppLogger().log('QuestService: Daily Quest f端r Stadt ${city.name} erfolgreich geladen: ${quest.qid}');
    return quest;
  }
  else {
    AppLogger().log('QuestService: Fehler beim Laden der Daily Quest - Status: ${response.statusCode}');
    throw Exception('Fehler beim Laden der Daily Quest: ${response.statusCode}');
  }
}
